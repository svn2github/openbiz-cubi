600 function substr_lr(&$str, $left, $right, &$startpos, $findfirst=false)
601 {
602  $pos0 = strpos($str, $left, $startpos);
603  if ($pos0 === false) return false;
604  $tmp = trim(substr($str,$startpos,$pos0-$startpos));
605  if ($findfirst && $tmp!="") return false;
606 
607  $posleft = $pos0+strlen($left);
608  while(true)
609  {
610  $pos1 = strpos($str, $right, $posleft);
611  if ($pos1 === false)
612  {
613  if (trim($right)=="")
614  {
615  $pos1 = strlen($str); // if right is whitespace
616  break;
617  }
618  else return false;
619  }
620  else
621  { // avoid \$right is found
622  if (substr($str,$pos1-1,1) == "\\") $posleft = $pos1+1;
623  else break;
624  }
625  }
626 
627  $startpos = $pos1 + strlen($right);
628  $retStr = substr($str, $pos0 + strlen($left), $pos1-$pos0-strlen($left));
629  return $retStr;
630 }
