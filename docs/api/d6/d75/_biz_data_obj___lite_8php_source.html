505  private function _getNumberRecords($db, $sql)
506  {
507  $has_subquery=false;
508  if(preg_match("/\(\s*?SELECT\s*?.+\)/si",$sql)){
509  $has_subquery=true;
510  }
511  if (preg_match("/^\s*SELECT\s+DISTINCT/is", $sql) || preg_match('/\s+GROUP\s+BY\s+/is',$sql))
512  {
513  // ok, has SELECT DISTINCT or GROUP BY so see if we can use a table alias
514  $rewritesql = preg_replace('/(\sORDER\s+BY\s.*)/is','',$sql);
515  $rewritesql = "SELECT COUNT(*) FROM ($rewritesql) _TABLE_ALIAS_";
516  }
517  elseif($has_subquery==false)
518  {
519  // now replace SELECT ... FROM with SELECT COUNT(*) FROM
520  $rewritesql = preg_replace('/\s*?SELECT\s.*?\s+FROM\s/is','SELECT COUNT(*) FROM ',$sql);
521  // Because count(*) and 'order by' fails with mssql, access and postgresql.
522  // Also a good speedup optimization - skips sorting!
523  $rewritesql = preg_replace('/(\sORDER\s+BY\s.*)/is','',$rewritesql);
524  }else{
525  $rewritesql = $sql;
526  }

531  $bindValueString = QueryStringParam::getBindValueString();
